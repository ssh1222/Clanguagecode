int* twoSum(int* nums, int numsSize, int target) {
    static int sum[100] = { 0 };
    for(int i = 0; i < numsSize; ++i)
    {
        for(int j = i+1; j < numsSize; ++j)
        {
            if(nums[i] + nums[j] == target)
            {
                sum[0] = i;
                sum[1] = j;
                return sum;
            }
        }
    }
    return 0;
}
